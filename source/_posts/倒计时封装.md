   export const useCount = () => {
  let timer = 0
  const count = ref(0)
  const start = (time = 60) => {
    if (count.value > 0) return
    //倒计时初始值
    count.value = time
    timer = setInterval(() => {
      count.value--
      if (count.value === 0) {
        // 到0退出
        clearInterval(timer)
      }
    }, 1000)
  }

  // 离开页面清除定时器
  onUnmounted(() => {
    clearInterval(timer)
  })

  return { count, start }
}
